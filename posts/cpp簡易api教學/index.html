void hello_world(const Request& req,Response& resp) {
  printf("%s [%s] %s, %s, %s %i \n", 
            ctime(&my_time),
            req.method.c_str(),
            req.path.c_str(),
            req.remote_addr.c_str(),
            req.version.c_str(),
            req.remote_port);
  resp.set_content("<html><h1>Hello world.</h1></html>","text/html");
}

int main() {
  Server svr;
  svr.Get("/api/hello", hello_world);
  svr.listen("0.0.0.0", 8000);
  system("pause");
  return 0;
}
